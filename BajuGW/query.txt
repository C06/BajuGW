CREATE TABLE User_Kategori (username VARCHAR (8) REFERENCES User (username) ON DELETE CASCADE ON UPDATE CASCADE, kategori_nama TEXT REFERENCES Kategori_List (kategori_nama) ON DELETE CASCADE ON UPDATE CASCADE, UNIQUE (username, kategori_nama));
CREATE TABLE Toko_Online_List (nama_toko TEXT PRIMARY KEY);
CREATE TABLE Kategori_Pakaian (username VARCHAR (8) REFERENCES User (username) ON DELETE CASCADE ON UPDATE CASCADE, kategori_nama TEXT REFERENCES Kategori_List (kategori_nama) ON DELETE CASCADE ON UPDATE CASCADE, item_id INTEGER REFERENCES Pakaian (item_id) ON DELETE CASCADE ON UPDATE CASCADE, UNIQUE (username, kategori_nama, item_id));
CREATE TABLE Connected_Store (username VARCHAR (8) REFERENCES User (username) ON DELETE CASCADE ON UPDATE CASCADE, nama_toko TEXT REFERENCES Toko_Online_List (nama_toko) ON DELETE NO ACTION ON UPDATE NO ACTION, UNIQUE (username, nama_toko));
CREATE TABLE User (username VARCHAR (8) PRIMARY KEY, data_wajah BLOB NOT NULL, email TEXT NOT NULL, password TEXT NOT NULL, salt TEXT UNIQUE NOT NULL, ukuran_badan NUMERIC, warna TEXT DEFAULT black_iron) WITHOUT ROWID;
CREATE TABLE Pakaian (item_id INTEGER PRIMARY KEY ON CONFLICT ROLLBACK AUTOINCREMENT, name TEXT DEFAULT Unidentified_Object, favorit BOOLEAN, ukuran_pakaian NUMERIC, warna TEXT, merek TEXT, file_pakaian BLOB NOT NULL, username VARCHAR (8) REFERENCES User (username) ON DELETE CASCADE ON UPDATE CASCADE);
CREATE TABLE Kategori_List (kategori_nama TEXT PRIMARY KEY) WITHOUT ROWID;
